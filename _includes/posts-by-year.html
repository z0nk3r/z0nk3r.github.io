{% comment %} Groups include.posts by year and renders each group as a year-heading plus a card grid - the shared shape behind blog.md, index.md, and _layouts/author.html, which used to each duplicate this loop verbatim. Usage: {% include posts-by-year.html posts=some_post_list %} {% endcomment %} {% assign posts_by_year = include.posts | group_by_exp: "post", "post.date | date: '%Y'" %} {% for year_group in posts_by_year %}

{{ year_group.name }}

    {% for post in year_group.items %} {% include post-card.html post=post %} {% endfor %}
{% endfor %}
